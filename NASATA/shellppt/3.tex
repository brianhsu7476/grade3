\begin{frame}{Condition}
The spaces in the following syntaxes are important!
\begin{multicols}{2}
\begin{itemize}
\only<1>{
\item Expression: \t{[[ expr ]]}
\item Boolen operations:
\begin{itemize}
\item \t{[[ ! expr ]]}: not
\item \t{[[ expr1 ]] \&\& [[ expr2 ]]}: and
\item \t{[[ expr1 ]] || [[ expr2 ]]}: or
\end{itemize}
\item Files:
\begin{itemize}
\item \t{[[ -e FILE ]]}: exists
\item \t{[[ -f FILE ]]}: is file
\item \t{[[ -d FILE ]]}: is directory
\item \t{[[ -L FILE ]]}, \t{[[ -h FILE ]]}: is symbolic link
\end{itemize}
}\only<2>{
\item Strings:
\begin{itemize}
\item \t{[[ -z str ]]}: is empty string
\item \t{[[ -n str ]]}: is non-empty string
\item \t{[[ str1 == str2 ]]}: is equal
\item \t{[[ str1 != str2 ]]}: is unequal
\item \t{[[ str =\sm\ regex ]]}: \t{str} satisfies \t{regex}
\end{itemize}
\item Numerical comparison:
\begin{itemize}
\item \t{[[ a -eq b ]]}: $a=b$
\item \t{[[ a -ne b ]]}: $a\neq b$
\item \t{[[ a -lt b ]]}: $a<b$
\item \t{[[ a -le b ]]}: $a\leq b$
\item \t{[[ a -gt b ]]}: $a>b$
\item \t{[[ a -ge b ]]}: $a\geq b$
\end{itemize}
}
\end{itemize}
\end{multicols}
\end{frame}
